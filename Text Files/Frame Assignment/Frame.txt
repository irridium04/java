    void CalcTemp()
    {
        try
        {
            float inTemp = Float.parseFloat(jTxtTempIn.getText());
            float outTemp = 0;
            if(jrbFToC.isSelected())
            {
                outTemp = (inTemp - 32) * (5f / 9f);
            }
            else if(jrbCtoF.isSelected())
            {
                outTemp = (inTemp * (9f / 5f)) + 32;
            }

            jLblTempAns.setText(String.valueOf(outTemp));
        }
        catch(Exception e)
        {
            jLblTempAns.setText("Error");
        }
    }
    
    void CalcTriArea()
    {
        try
        {
            float b = Float.parseFloat(jTxtTriBase.getText()); 
            float h = Float.parseFloat(jTxtTriHeight.getText());

            float a = .5f * b * h;

            jLblTriAns.setText("Area: " + a);
        }
        catch(Exception e)
        {
            jLblTriAns.setText("Error");
        }
    }
    
    void CalcTrapArea()
    {
        try
        {
            float b1 = Float.parseFloat(jTxtTrapBase1.getText()); 
            float b2 = Float.parseFloat(jTxtTrapBase2.getText()); 
            float h = Float.parseFloat(jTxtTrapHeight.getText());

            float a = ((b1 + b2) / 2f) * h;

            jLblTrapAns.setText("Area: " + a);
        }
        catch(Exception e)
        {
            jLblTrapAns.setText("Error");
        }
    }
